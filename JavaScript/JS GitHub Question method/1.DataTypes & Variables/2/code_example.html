2я╕ПтГг Checking Array using Array.isArray()

<script>
let numbers = [1, 2, 3];
let name = "Rocky";

if (Array.isArray(numbers)) {
  console.log("numbers is an array");
}

if (!Array.isArray(name)) {
  console.log("name is not an array");
}
</script>

Output:

numbers is an array
name is not an array
